var nums = [0,1,0,3,12,0,5,0,0,0,0,0];

var moveZeroes = function(nums) {
    var x = nums.indexOf(0);
    console.log(x);
    var count = 0;
    while(x!=-1){
        count++;
        nums.splice(x,1);
        x = nums.indexOf(0);
        console.log(x);
    }
    while(count){
        count--;
        nums.push(0);
    }
    console.log(nums);
};
